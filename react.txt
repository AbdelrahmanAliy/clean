import { useEffect, useState } from "react";

function App() {
  const [alpha, setAlpha] = useState([]);

  useEffect(() => {
    fetch("/api/alpha")
      .then(res => res.json())
      .then(setAlpha);
  }, []);

  return (
    <div>
      {alpha.map(item => (
        <p key={item.id}>{item.name}</p>
      ))}
    </div>
  );
}

export default App;
